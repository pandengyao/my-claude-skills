# GDP2 SQL API Generator

## Overview

Generate complete CRUD API interfaces from database table structures following the GDP2 framework MVC architecture pattern.

## Workflow

### 1. Extract Table Schema

Use the shell script to retrieve table structure:

```bash
# Using default connection (yc_union database)
.comate/skills/gdp2-sql-api/scripts/get_table_schema.sh table_name

# Using custom connection
.comate/skills/gdp2-sql-api/scripts/get_table_schema.sh \
  -h localhost -P 3306 -u root -p password -d mydb table_name

# Save to file
.comate/skills/gdp2-sql-api/scripts/get_table_schema.sh \
  -o schema.json table_name
```

package dao

import "gorm.io/gorm"

type TableName struct {
	ID        int64  `gorm:"column:id;primary_key;AUTO_INCREMENT" json:"id"`

	UpdateTime int   `gorm:"column:update_time;default:0;NOT NULL" json:"update_time"`

func (t *TableName) TableName() string {
	return "table_name"
}

#### DAO Functions (model/dao/table_name_dao.go)
```go
package dao

import (
	"context"
	"gorm.io/gorm"
)

func GetTableNameByID(ctx context.Context, id int64) (*TableName, error) {
	var result TableName
	err := GetGormYcUnionConn().WithContext(ctx).
		Where("id = ?", id).
		Where("status = ?", 1).
		First(&result).Error
	return &result, err
}

func CreateTableName(ctx context.Context, entity *TableName) error {
	return GetGormYcUnionConn().WithContext(ctx).Create(entity).Error
}

func UpdateTableName(ctx context.Context, id int64, updates map[string]interface{}) error {
	return GetGormYcUnionConn().WithContext(ctx).
		Model(&TableName{}).
		Where("id = ?", id).
		Updates(updates).Error
}

func DeleteTableName(ctx context.Context, id int64) error {
	return GetGormYcUnionConn().WithContext(ctx).
		Model(&TableName{}).
		Where("id = ?", id).
		Update("status", 0).Error
}

func ListTableName(ctx context.Context, conditions map[string]interface{}, limit, offset int) ([]*TableName, error) {
	var results []*TableName
	db := GetGormYcUnionConn().WithContext(ctx)

	for key, value := range conditions {
		db = db.Where(key+" = ?", value)
	}

	db = db.Where("status = ?", 1)
	if limit > 0 {
		db = db.Limit(limit)
	}
	if offset > 0 {
		db = db.Offset(offset)
	}

	err := db.Find(&results).Error
	return results, err
}
```

#### Service (model/service/table_name_service.go)
```go
package service

import (
	"context"
	"your-project/model/dao"
	"your-project/library/types/request"
	"your-project/library/types/response"
)

func CreateTableName(ctx context.Context, req *request.CreateTableNameReq) (*response.TableNameResp, error) {
	entity := &dao.TableName{
		FieldName: req.FieldName,
	}

	if err := dao.CreateTableName(ctx, entity); err != nil {
		return nil, err
	}

	return &response.TableNameResp{
		ID:        entity.ID,
		FieldName: entity.FieldName,
	}, nil
}

func UpdateTableName(ctx context.Context, req *request.UpdateTableNameReq) error {
	updates := map[string]interface{}{
		"field_name": req.FieldName,
	}
	return dao.UpdateTableName(ctx, req.ID, updates)
}

func DeleteTableName(ctx context.Context, id int64) error {
	return dao.DeleteTableName(ctx, id)
}

func QueryTableName(ctx context.Context, req *request.QueryTableNameReq) (*response.TableNameResp, error) {
	entity, err := dao.GetTableNameByID(ctx, req.ID)
	if err != nil {
		return nil, err
	}

	return &response.TableNameResp{
		ID:        entity.ID,
		FieldName: entity.FieldName,
	}, nil
}

func ListTableName(ctx context.Context, req *request.ListTableNameReq) ([]*response.TableNameResp, int64, error) {
	conditions := make(map[string]interface{})
	if req.FieldName != "" {
		conditions["field_name"] = req.FieldName
	}

	offset := (req.Page - 1) * req.PageSize
	entities, err := dao.ListTableName(ctx, conditions, req.PageSize, offset)
	if err != nil {
		return nil, 0, err
	}

	resp := make([]*response.TableNameResp, 0, len(entities))
	for _, entity := range entities {
		resp = append(resp, &response.TableNameResp{
			ID:        entity.ID,
			FieldName: entity.FieldName,
		})
	}

	return resp, int64(len(resp)), nil
}
```

## Project Structure Reference

Follow these paths when generating files:
- **Router**: `servers/httpserver/routers/`
- **Controller**: `servers/httpserver/controller/`
- **Service**: `model/service/`
- **DAO**: `model/dao/`
- **Request**: `library/types/request/`
- **Response**: `library/types/response/`

## Database Connection Functions

Use appropriate connection function based on database:
- `GetGormBackendConn()` - 商家后台库
- `GetGormYcUnionConn()` - 运营库
- `GetGormYongcheConn()` - 打车主库
- `GetGormYcUserConn()` - 用户库
- `GetGormSettlementConn()` - 结算库
- `GetGormKefuConn()` - 客服库

## Naming Conventions

- **File names**: snake_case (e.g., `table_name.go`)
- **Struct names**: PascalCase (e.g., `TableName`)
- **Function names**: PascalCase for exported (e.g., `CreateTableName`)
- **Database columns**: snake_case (e.g., `field_name`)
- **JSON fields**: snake_case (e.g., `field_name`)
